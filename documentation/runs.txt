in the venv
# 1) Go to the workspace
cd ~/ros2_ws

# 2) Source ROS 2 Jazzy
source /opt/ros/jazzy/setup.bash

# 3) Activate your virtualenv (the one with numpy<2 + ultralytics)
source ~/yolo_env/bin/activate

# 4) Prevent Python from loading user-site packages (~/.local) which had numpy 2.2.6
export PYTHONNOUSERSITE=1

# (optional but you did it earlier) Make colcon use the venv python
# export COLCON_PYTHON_EXECUTABLE=$(which python3)

# 5) Clean previous build artifacts
rm -rf build install log

# 6) Build using the venv python so the generated entrypoints use the venv shebang
python3 -m colcon build --symlink-install

# 7) Source the workspace overlay
source install/setup.bash

# 8) Run the node
ros2 run cone_detector_node detector
